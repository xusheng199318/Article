### 程序的机器级表示（上）

mstore.c

~~~c
long mult2(long, long);

void mulstore(long x, long y, long *dest) {
        long t = mult2(x, y);
        *dest = t;
}

~~~

通过`gcc -Og -S mstore.c`命令生成如下汇编代码

~~~assembly
mulstore:
        pushq   %rbx
        movq    %rdx, %rbx
        call    mult2
        movq    %rax, (%rbx)
        popq    %rbx
        ret
       

~~~

通过`gcc -Og -c mstore.c`产生mstore.o文件，通过`objdump -d mstore.o`查看该文件

~~~assembly

   0:	53                   	push   %rbx

   9:	48 89 03             	mov    %rax,(%rbx)
   c:	5b                   	pop    %rbx
   d:	c3                   	retq   

~~~
